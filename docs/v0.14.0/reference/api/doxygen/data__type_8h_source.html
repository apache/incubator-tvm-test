249 inline int GetVectorBytes(DataType dtype) {
250  int data_bits = dtype.bits() * dtype.lanes();
251  // allow bool to exist

346  if (t.bits == 1 && t.lanes == 1 && t.code == kDLUInt) {

377 inline DLDataType String2DLDataType(std::string s) {
378  DLDataType t;
379  // handle void type
380  if (s.length() == 0 || s == "void") {
381  t = DataType::Void();
382  return t;
383  }
384  t.bits = 32;
385  t.lanes = 1;
386  const char* scan;
387  if (s.substr(0, 3) == "int") {
388  t.code = kDLInt;
389  scan = s.c_str() + 3;
390  } else if (s.substr(0, 4) == "uint") {
391  t.code = kDLUInt;
392  scan = s.c_str() + 4;
393  } else if (s.substr(0, 5) == "float") {
394  t.code = kDLFloat;
395  scan = s.c_str() + 5;
396  } else if (s.substr(0, 6) == "handle") {
397  t.code = kTVMOpaqueHandle;
398  t.bits = 64; // handle uses 64 bit by default.
399  scan = s.c_str() + 6;
400  } else if (s == "bool") {
401  t.code = kDLUInt;
402  t.bits = 1;
403  t.lanes = 1;
404  return t;
405  } else if (s.substr(0, 6) == "bfloat") {
406  t.code = DataType::kBFloat;
407  scan = s.c_str() + 6;
408  } else if (s.substr(0, 10) == "e4m3_float") {
409  t.code = DataType::kE4M3Float;
410  scan = s.c_str() + 10;
411  } else if (s.substr(0, 10) == "e5m2_float") {
412  t.code = DataType::kE5M2Float;
413  scan = s.c_str() + 10;
414  } else if (s.substr(0, 6) == "custom") {
415  t.code = ParseCustomDatatype(s, &scan);
416  } else {
417  scan = s.c_str();
418  LOG(FATAL) << "unknown type " << s;
419  }
420  char* xdelim; // emulate sscanf("%ux%u", bits, lanes)
421  uint8_t bits = static_cast<uint8_t>(strtoul(scan, &xdelim, 10));
422  if (bits != 0) t.bits = bits;
423  char* endpt = xdelim;
424  if (*xdelim == 'x') {
425  t.lanes = static_cast<uint16_t>(strtoul(xdelim + 1, &endpt, 10));
426  }
427  ICHECK(endpt == s.c_str() + s.length()) << "unknown type " << s;
428  return t;
429 }
